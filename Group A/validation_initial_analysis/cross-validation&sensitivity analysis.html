# Adjust all thresholds by a buffer to account for possible imputation
buffer_percentage = 1.1  # 10% buffer for all values
ghg_intensity_benchmarks = {
    'platinum': 0.1,
    'goldplus': 0.12,
    'gold': 0.15,
    'certified': 0.2
}
# Adjusted function assuming potential imputation
def validate_ghg_intensity(row):
    green_mark_level = row['Award']
    ghg_intensity = row['GHG_Intensity']
    
    # Check if Green Mark level is missing
    if pd.isna(green_mark_level):
        return 'No Green Mark'
    
    # Apply buffer to the benchmark threshold
    threshold = ghg_intensity_benchmarks.get(green_mark_level, None)
    if threshold:
        threshold *= buffer_percentage  # Apply the buffer uniformly
    
        # Perform validation
        if ghg_intensity <= threshold:
            return 'Pass'
        else:
            return 'Fail'
    return 'No Green Mark'

# Define the GHG Intensity benchmarks for Green Mark levels
ghg_intensity_benchmarks = {
    'platinum': 0.1,
    'goldplus': 0.12,
    'gold': 0.15,
    'certified': 0.2
}

# List of dataframes
dataframes = [merged_df1, merged_df2, merged_df3, merged_df4, merged_df5]
match_rates = []

# Validation function
def validate_ghg_intensity(row):
    green_mark_level = row['Award']
    ghg_intensity = row['GHG_Intensity']
    
    # Check for missing Green Mark levels
    if pd.isna(green_mark_level):
        return None
    
    # Get threshold
    green_mark_level = green_mark_level.lower()
    threshold = ghg_intensity_benchmarks.get(green_mark_level)
    
    # Return True for Pass, False for Fail
    if threshold:
        return ghg_intensity <= threshold
    return None

# Function to calculate match rate
def calculate_match_rate(df):
    # Apply validation
    df['Validation Result'] = df.apply(validate_ghg_intensity, axis=1)
    
    # Determine expected result based on Green Mark level (assume all should Pass for simplicity)
    df['Expected Result'] = df['Award'].notna()  # True if there is an award
    
    # Calculate match rate
    correct_matches = (df['Validation Result'] == df['Expected Result']).sum()
    total_valid_rows = df['Expected Result'].notna().sum()
    
    match_rate = correct_matches / total_valid_rows if total_valid_rows > 0 else 0
    return match_rate

# Calculate match rate for each dataframe
for idx, df in enumerate(dataframes, start=1):
    match_rate = calculate_match_rate(df)
    match_rates.append((f'merged_df{idx}', match_rate))
    print(f'merged_df{idx} Match Rate: {match_rate:.2%}')

# Find the best matching dataframe
best_df, best_match_rate = max(match_rates, key=lambda x: x[1])
print(f"The best matching dataset is {best_df} with a match rate of {best_match_rate:.2%}")

import matplotlib.pyplot as plt
import seaborn as sns

# Set up visualization style
sns.set(style="whitegrid")

# Create a combined bar plot for validation results across all datasets
def combined_validation_plot(dataframes):
    result_counts = []
    
    # Collect validation results for each dataframe
    for idx, df in enumerate(dataframes, start=1):
        validation_counts = df['Validation Result'].value_counts()
        result_counts.append(pd.DataFrame({
            'Result': validation_counts.index,
            'Count': validation_counts.values,
            'Dataset': f'merged_df{idx}'
        }))
    
    # Concatenate all results
    result_df = pd.concat(result_counts)
    
    # Create the bar plot
    plt.figure(figsize=(12, 8))
    sns.barplot(x='Dataset', y='Count', hue='Result', data=result_df, palette='viridis')
    
    # Set labels and title
    plt.xlabel('Dataset', fontsize=14)
    plt.ylabel('Count', fontsize=14)
    plt.title('GHG Intensity Validation Results Across Datasets', fontsize=16)
    
    # Show plot
    plt.show()

# Plot combined validation results

            adjusted_df['Scope3'] = final_df['Scope3'] * factor

            adjusted_df['Scope3'] = final_df['Scope3'] * factor
